Create a list:


old_list = [item_1, item_2, item_3]

# or from a range:
                        
new_list = list(range(start, stop, stepover))
                        
# stepover - is "take every N-th element from the start"

Create an array:


// Old way:

const array_name = new Array(item1, item2, item3);

// It is not recommended, because array_name = new Array(10); 
// This will create an array with 10 undefined elements.
                        
// New way:

const array_name = [item1, item2, item3];

// or with an empty array and indexes:
                        
const array_name = [];
array_name[0] = item1;
array_name[1] = item2;
array_name[2] = item3;

                        new_list = old_list[-3:] # everything that is at the right of the item with index -3
                        new_list = old_list[:-3] # everything that is at the left of the item with index -3

                        # Also you can create a reversed list:

                        new_list = old_list[::-1]
